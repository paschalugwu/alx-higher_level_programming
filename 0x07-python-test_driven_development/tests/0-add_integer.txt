def test_add_integer():
    """
    Test the add_integer function
    """
    # Test case 1: Adding two integers
    assert add_integer(1, 2) == 3

    # Test case 2: Adding positive and negative integers
    assert add_integer(100, -2) == 98

    # Test case 3: Adding one integer and the default value
    assert add_integer(2) == 100

    # Test case 4: Adding a float and an integer
    assert add_integer(100.3, -2) == 98

    # Test case 5: Adding an integer and a non-integer
    try:
        add_integer(4, "School")
        assert False, "Exception not raised"
    except TypeError as e:
        assert str(e) == "a must be an integer"

    # Test case 6: Adding None
    try:
        add_integer(None)
        assert False, "Exception not raised"
    except TypeError as e:
        assert str(e) == "a must be an integer"
